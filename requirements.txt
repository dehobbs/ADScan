# ADScan - Python Dependencies
# Install with: pip install -r requirements.txt
#
# Most packages are pinned to exact versions (==) that have been tested together.
# Upgrade with care — run the full test suite after any version bump.
# Exception: ldap3 is pinned to >=2.10.0rc1 (minimum for LDAPS channel binding support).
#
# Compatibility notes:
#   - certipy-ad 5.0.4 hard-requires pyasn1==0.4.8 (see its setup.py).
#     Do NOT upgrade pyasn1 without first upgrading certipy-ad to 5.x
#     (which requires Python >=3.12 and carries breaking API changes).
#   - pyasn1-modules 0.4.1 is the newest release whose dependency range
#     (pyasn1>=0.4.6,<0.7.0) is satisfied by pyasn1==0.4.8.
#   - impacket 0.13.0 is compatible with certipy-ad 5.0.4.
#   - cryptography 42.0.8 is pinned by certipy-ad 5.0.4 (~=42.0.8)
#     and fits within pyOpenSSL 25.1.0's range (>=41.0.5,<46).
#   - pyOpenSSL 25.1.0: highest release compatible with cryptography<45.
#     26.0.0 requires >=46; 25.2.0+ requires >=45.

# LDAP/LDAPS connectivity (required for LDAP-based checks)
ldap3>=2.10.0rc1  # 2.10.x required for LDAPS channel binding (TLS_CHANNEL_BINDING) support

# SMB connectivity and NTLM authentication (required for SMB checks and pass-the-hash)
impacket==0.13.0

# ---------------------------------------------------------------------------
# Certipy-AD - ADCS / PKI vulnerability scanner (required for check_adcs.py)
# Requires Python >=3.12.
# GitHub: https://github.com/ly4k/Certipy
# ---------------------------------------------------------------------------
certipy-ad==5.0.4

# Cryptography toolkit
# 42.0.8 is pinned by certipy-ad 5.0.4 (~=42.0.8)
cryptography==42.0.8

# ASN.1 parsing
# pyasn1 0.4.8 is hard-required by certipy-ad 5.0.4 — do not change independently
pyasn1==0.4.8
# pyasn1-modules 0.4.1: latest release compatible with pyasn1>=0.4.6,<0.7.0
pyasn1-modules==0.4.1

# DNS resolution
dnspython==2.8.0

# HTTP/HTTPS requests (used by certipy for web enrollment checks)
requests==2.32.5

# OpenSSL Python bindings
# 25.1.0 is the highest version whose cryptography range (>=41.0.5,<46) includes 42.0.8
pyOpenSSL==25.1.0

# Safe XML parsing — prevents XML bomb / entity expansion attacks (CWE-20 / Bandit B314)
defusedxml==0.7.1

# Word document export (optional -- only needed for --format docx)
python-docx>=1.1.2

# Image rendering for DOCX Command & Output blocks (optional -- only needed for --format docx)
Pillow>=10.0.0
