# ADScan - Python Dependencies
# Install with: pip install -r requirements.txt
#
# Most packages are pinned to exact versions (==) that have been tested together.
# Upgrade with care — run the full test suite after any version bump.
# Exception: ldap3 is pinned to >=2.10.0rc1 (minimum for LDAPS channel binding support).
#
# Compatibility notes:
#   - certipy-ad is now installed as an isolated tool via `uv tool install`
#     (see lib/tools.py), so its dependency pins no longer conflict here.
#   - pyasn1-modules 0.4.1 is the newest release whose dependency range
#     (pyasn1>=0.4.6,<0.7.0) is satisfied by pyasn1==0.4.8.
#   - cryptography 44.0.3 fits within pyOpenSSL 25.1.0's range (>=41.0.5,<46).
#   - pyOpenSSL 25.1.0: highest release compatible with cryptography<45.
#     26.0.0 requires >=46; 25.2.0+ requires >=45.

# LDAP/LDAPS connectivity (required for LDAP-based checks)
ldap3>=2.10.0rc1  # 2.10.x required for LDAPS channel binding (TLS_CHANNEL_BINDING) support

# SMB connectivity and NTLM authentication (required for SMB checks and pass-the-hash)
impacket==0.13.0

# ---------------------------------------------------------------------------
# External CLI tools — installed in isolated venvs via uv (not pip):
#   uv tool install certipy-ad     # ADCS/PKI scanner (v5.0.4+, Python 3.12+)
#   uv tool install netexec        # SMB signing/SMBv1 detection (nxc command)
#   uv tool install git+https://github.com/garrettfoster13/pre2k-TS.git  # Pre-Windows 2000 computer account tester
# Or run: python adscan.py --setup-tools

# ---------------------------------------------------------------------------
# Cryptography toolkit
# 44.0.3 satisfies pyOpenSSL 25.1.0 (>=41.0.5,<46) and impacket 0.13.0
cryptography==44.0.3

# ASN.1 parsing (required by impacket)
pyasn1==0.4.8
# pyasn1-modules 0.4.1: latest release compatible with pyasn1>=0.4.6,<0.7.0
pyasn1-modules==0.4.1

# DNS resolution
dnspython==2.8.0

# HTTP/HTTPS requests (used by impacket)
requests==2.32.5

# OpenSSL Python bindings
# 25.1.0 is the highest version whose cryptography range (>=41.0.5,<46) includes 44.0.3
pyOpenSSL==25.1.0

# Safe XML parsing — prevents XML bomb / entity expansion attacks (CWE-20 / Bandit B314)
defusedxml==0.7.1

# Word document export (required for --format docx)
python-docx>=1.1.2

# Image rendering for DOCX Command & Output blocks (required for --format docx)
Pillow>=10.0.0
